```
# make virtualenv
virtualenv venv

# activate venv
venv\Scripts\activate  # windows
. venv/bin/activate  # Linux, OS X

# install requirements
pip install -r requirements.txt

# run app
python main.py

# run unittests
python -m unittest */test.py
```
